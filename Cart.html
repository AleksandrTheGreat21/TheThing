function getCookie(name) { var cookieArr = document.cookie.split("; "); for (var i = 0; i < cookieArr.length; i++) { var cookiePair = cookieArr[i].split("="); if (name === cookiePair[0]) { return cookiePair[1]; } } return null; } window.onload = function() { var x = getCookie("x"); var y = getCookie("y"); if (x !== null && y !== null) { // Do something with x and y, for example, display them on the page document.getElementById("x_value").innerText = x; document.getElementById("y_value").innerText = y; } else { // Cookies not found or not set, handle accordingly console.log("Cookies not found"); } }
